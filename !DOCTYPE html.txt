    <script>
        const canvas = document.getElementById('game');
        const ctx = canvas.getContext('2d');
        const grid = 20;
        let snake = [{x: 160, y: 200}];
        let dx = grid, dy = 0;
        let food = spawnFood();
        let score = 0;
        let gameLoop;
        let paused = true;
        let gameSpeed = 200;
        // 你的奖励图片链接，已全部替换
        const awardImgList = [
            "https://s3.bmp.ovh/2026/05/10/f10mVX9v.jpg",
            "https://s3.bmp.ovh/2026/05/10/oEMFxfrI.jpg",
            "https://s3.bmp.ovh/2026/05/10/00ePSVJ9.jpg",
            "https://s3.bmp.ovh/2026/05/10/WupSW8Qq.jpg",
            "https://s3.bmp.ovh/2026/05/09/WssOSeD9.jpg",
            "https://s3.bmp.ovh/2026/05/09/UmiUpCt0.jpg",
            "https://s3.bmp.ovh/2026/05/09/hHule2EJ.jpg",
            "https://s3.bmp.ovh/2026/05/09/s1fIXVfr.jpg",
            "https://s3.bmp.ovh/2026/05/09/Nb3DqifC.jpg",
            "https://s3.bmp.ovh/2026/05/09/9VChLzhV.jpg",
            "https://s3.bmp.ovh/2026/05/09/3Wh2daVr.jpg",
            "https://s3.bmp.ovh/2026/05/09/N6zhDVAS.png",
            "https://s3.bmp.ovh/2026/05/09/ql3jUp9R.webp"
        ];
        // 记录吃到的豆子数量
        let eatCount = 0;
        // 音频
        const bgMusic = document.getElementById('bgMusic');
        const eatSound = document.getElementById('eatSound');
        const musicBtn = document.getElementById('musicBtn');
        const soundBtn = document.getElementById('soundBtn');
        let musicOn = true;
        let soundOn = true;
        // 奖励弹窗元素
        const awardMask = document.getElementById('awardMask');
        const awardImg = document.getElementById('awardImg');
        const closeAwardBtn = document.getElementById('closeAwardBtn');
        // 关闭奖励弹窗
        closeAwardBtn.onclick = () => {
            awardMask.style.display = 'none';
            paused = false;
            document.getElementById('startBtn').textContent = '暂停';
        };
        // 菜单元素
        const settingBtn = document.getElementById('settingBtn');
        const menuBox = document.getElementById('menuBox');
        const difficultyBtns = document.querySelectorAll('.diff-btn');
        settingBtn.addEventListener('click', () => {
            menuBox.style.display = menuBox.style.display === 'block' ? 'none' : 'block';
        });
        document.addEventListener('click', (e) => {
            if(!settingBtn.contains(e.target) && !menuBox.contains(e.target)){
                menuBox.style.display = 'none';
            }
        });
        // 随机选一张奖励图
        function getRandomAwardImg(){
            let randomIndex = Math.floor(Math.random() * awardImgList.length);
            return awardImgList[randomIndex];
        }
        // 弹出奖励图片
        function showAward() {
            paused = true;
            // 随机切换图片
            awardImg.src = getRandomAwardImg();
            awardMask.style.display = 'flex';
        }
        function spawnFood() {
            let x = Math.floor(Math.random() * (canvas.width / grid)) * grid;
            let y = Math.floor(Math.random() * (canvas.height / grid)) * grid;
            if (snake.some(p => p.x === x && p.y === y)) return spawnFood();
            return {x, y};
        }
        function draw() {
            ctx.fillStyle = '#000';
            ctx.fillRect(0, 0, canvas.width, canvas.height);
            ctx.fillStyle = '#0f0';
            snake.forEach(part => ctx.fillRect(part.x, part.y, grid-1, grid-1));
            ctx.fillStyle = '#f00';
            ctx.fillRect(food.x, food.y, grid-1, grid-1);
        }
        function playEatSound() {
            if (soundOn) {
                eatSound.currentTime = 0;
                eatSound.play().catch(err => {});
            }
        }
        function move() {
            if (paused) return;
            let head = {x: snake[0].x + dx, y: snake[0].y + dy};
            if (head.x < 0 || head.x >= canvas.width || head.y < 0 || head.y >= canvas.height || snake.some(p => p.x === head.x && p.y === head.y)) {
                bgMusic.pause();
                alert("游戏结束！得分：" + score);
                reset();
                return;
            }
            snake.unshift(head);
            if (head.x === food.x && head.y === food.y) {
                score += 10;
                eatCount++; 
                document.getElementById('score').textContent = "Score: " + score;
                playEatSound();
                food = spawnFood();
                // 每吃10个豆 随机弹一张你的奖励图
                if(eatCount % 10 === 0){
                    showAward();
                }
            } else {
                snake.pop();
            }
        }
        function reset() {
            clearInterval(gameLoop);
            gameLoop = null;
            snake = [{x: 160, y: 200}];
            dx = grid; dy = 0;
            score = 0;
            eatCount = 0; 
            paused = true;
            document.getElementById('score').textContent = "Score: 0";
            document.getElementById('startBtn').textContent = "开始";
            food = spawnFood();
            draw();
            bgMusic.pause();
            bgMusic.currentTime = 0;
            awardMask.style.display = 'none';
        }
        difficultyBtns.forEach(btn => {
            btn.addEventListener('click', () => {
                if (paused) {
                    difficultyBtns.forEach(b => b.classList.remove('active'));
                    btn.classList.add('active');
                    gameSpeed = Number(btn.dataset.speed);
                }
            });
        });
        musicBtn.addEventListener('click', () => {
            musicOn = !musicOn;
            musicBtn.classList.toggle('active');
            musicBtn.innerText = musicOn ? '开启' : '关闭';
            if (musicOn) {
                if (!paused) bgMusic.play().catch(err => {});
            } else {
                bgMusic.pause();
            }
        });
        soundBtn.addEventListener('click', () => {
            soundOn = !soundOn;
            soundBtn.classList.toggle('active');
            soundBtn.innerText = soundOn ? '开启' : '关闭';
        });
        document.getElementById('startBtn').onclick = () => {
            paused = !paused;
            document.getElementById('startBtn').textContent = paused ? "开始" : "暂停";
            if (paused) {
                clearInterval(gameLoop);
                gameLoop = null;
                bgMusic.pause();
            } else {
                if (musicOn) bgMusic.play().catch(err => {});
                if (!gameLoop) gameLoop = setInterval(() => { move(); draw(); }, gameSpeed);
            }
        };
        document.getElementById('resetBtn').onclick = reset;
        document.getElementById('up').onclick = () => { if (dy !== grid) { dx = 0; dy = -grid; } };
        document.getElementById('down').onclick = () => { if (dy !== -grid) { dx = 0; dy = grid; } };
        document.getElementById('left').onclick = () => { if (dx !== grid) { dx = -grid; dy = 0; } };
        document.getElementById('right').onclick = () => { if (dx !== -grid) { dx = grid; dy = 0; } };
        draw();
    </script>
</body>
